import pandas as pd
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv("meta_data_indicator.csv")

plt.xlabel('Subdomain')

# Q7: Flag quantitative descriptions (example logic)
df['Type'] = df['Indicator description'].apply(
    lambda x: 'Quantitative' if any(word in x.lower() for word in ['count', 'rate', '%']) else 'Qualitative'
